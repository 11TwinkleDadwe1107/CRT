#include <stdio.h>
int main() {
    int arr[10], i, sum=0;
    for(i=0; i<10; i++)
    {
        printf("enter a value of array arr[%d]:",i);
        scanf("%d",&arr[i]);
        sum+=arr[i];
    }
    printf("sum=%d",sum);
    return 0;
}

FIND EVEN&ODD ELEMENTS OF ARRAY.

#include <stdio.h>
int main()
{
    int arr[10], i, even=0, odd=0;
    for(i=0; i<10; i++)
    {
        printf("enter a value of array arr[%d]:",i);
        scanf("%d",&arr[i]);
        if(arr[i]%2==0)
        even++;
        else
        odd++;
    }
    printf("even=%d and odd=%d",even,odd);
    return 0;
}



FIND SMALLEST & LARGEST NUMBER IN ARRAY.

#include <stdio.h>
int main()
{
    int arr[10]={2,94,3,56,675,32,2,34,21,8};
    int small, large, i;
    small=large=arr[0];
    for(i=0; i<10; i++)
    {
        if(arr[i]<small)
        small=arr[i];
        if(arr[i]>large)
        large=arr[i];
    }
    printf("small=%d and large=%d",small, large);
    return 0;
}


TO INITIALISE A 2D ARRAY AND DISPLAY VALUES

#include <stdio.h>
int main()
{
    int matrix[4] [3]={
        {2,94,3},{56,65,32},{2,34,21}, {65,87,4}
 
    };
    for(int i=0; i<4; i++)
    {
        for(int j=0; j<3; j++)
        {
         printf("matrix[%d] [%d]",i,j);
         printf("%d\t", matrix[i][j]);
    }
    printf("\n");
    }
    return 0;
}
